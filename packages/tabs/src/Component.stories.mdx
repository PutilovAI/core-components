import { Meta, Story, Props, Preview, Title } from '@storybook/addon-docs/blocks';
import { boolean, select } from '@storybook/addon-knobs';

import { PrimaryTabsHeader } from './components/primary-tabs-header';
import { SecondaryTabsHeader } from './components/secondary-tabs-header';

import { name, version } from '../package.json';

export const titles = Array(7).fill(0).map((_, i) => ({ title: `Таб ${i + 1}`, id: i + 1 }));


<Meta title='Компоненты|Tabs' />

<Title>
    Tabs ({name}@{version})
</Title>

## Описание

Компоненты для построения табов.

### PrimaryTabsHeader

<Story name='PrimaryTabsHeader'>
    {React.createElement(() => {
        const [selected, setSelected] = React.useState(1);
        const handleChange = (event, { selected }) => setSelected(selected);
        return (
            <PrimaryTabsHeader
                titles={titles}
                selected={selected}
                scrollable={boolean('scrollable', false)}
                gaps={select('gaps', ['default', 'wide'], 'default')}
                onChange={handleChange}
            />
        );
    })}
</Story>
